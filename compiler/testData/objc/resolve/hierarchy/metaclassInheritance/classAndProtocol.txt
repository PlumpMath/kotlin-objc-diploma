    public class object <class-object-for-BaseClass> {

    }
}

public trait BaseProtocol {
    
    public class object <class-object-for-BaseProtocol> {

}

    public class object <class-object-for-Derived> : objc.BaseClass.<class-object-for-BaseClass> {
        public open fun derivedMethod() : jet.Unit
    }
}
